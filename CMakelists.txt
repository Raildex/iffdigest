cmake_minimum_required(VERSION 3.19)
project(iffdigest VERSION 0.3)
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_C_STANDARD 98)
add_library(iffdigest)
target_include_directories(iffdigest PUBLIC include)
target_include_directories(iffdigest PRIVATE src)
target_sources(iffdigest PRIVATE src/iffdigest.cc)
option(TEST_TYPE "Which Test to generate" "None")
if(TEST_TYPE STREQUAL "None")
elseif(TEST_TYPE STREQUAL "WAV")
elseif(TEST_TYPE STREQUAL "Dump")
endif()
